/* Variadic function — MISRA Rule 17.1 violation */
int sum_ints(int count, ...)
{
    va_list ap;
    int s = 0;
    int i;
    va_start(ap, count);
    for (i = 0; i < count; i++) {
        s += va_arg(ap, int);
    }
    va_end(ap);
    return s;
}

/* Recursion — MISRA Rule 17.2 violation */
int factorial(int n)
{
    if (n <= 1) {
        return 1;
    }
    return n * factorial(n - 1);
}

/* Dynamic allocation and goto — MISRA Rules 21.3 and 15.1 violations */
int* make_buffer(int n)
{
    int *p = (int *)malloc((size_t)n * sizeof(int));
    if (p == NULL) {
        goto fail; /* goto used for error handling */
    }
    return p;
fail:
    /* fall-through to return NULL */
    return NULL;
}

/* Dynamic memory free — MISRA Rule 21.3 violation in safety-critical paths */

/* If-else chain without final else; multiple early returns; uninitialised variable use — 9.1, 15.5, 15.7 */
